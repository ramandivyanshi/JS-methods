### camelize

function camelcase(str){
  var result="";
  for(var i=0;i<str.length;i++){
    var input=str[i];
    var temp=input.toLowerCase();
    if(i!=0){
      temp=temp.substr(0,1).toUpperCase()+temp.substr(1);
    }
    result+=temp;
  }
  return result;
}
